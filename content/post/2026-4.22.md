### ABC 451 E 代码

- 题面描述给给定一个距离矩阵：问是否存在树使得距离矩阵成立
- 结论是：如果存在一定是MST
- ![1776863971617](/post/image/2026-4.22/1776863971617.png)
- 可以根据这个题看下本人理解：一开始我们找到所有距离中最小的，那么必然要连接这条边。
- 然后归纳法：继续找所有距离次小的，此时有两种情况，如果最小距离是A-B,那么依就直接连接A-B ，否则如果是图中所示u-v-C的结构，那么显然最小边只能是v-C，不能是连接u-C
- 后面依旧同理，所以我们本质是在做一个MST。
- 求出MST，在check别的距离矩阵就可以了！

```cpp
#include <bits/stdc++.h>
typedef long long ll;
typedef long double db;
const ll mod=1e9+7;
const db pi=acos(-1);
const db eps=1e-10;
const int maxn=4000+10;
const int inf=1e7;
const ll INF=1e17;
using namespace std;
const char mv[4] = {'R', 'L', 'D', 'U'};
int n;
int A[maxn][maxn];
struct DSU{
    vector<int>f;
    vector<int>dis;
    vector<int>p;
    void init(int n) {
        dis.assign(n+1,0);
        p.assign(n+1,0);
        f.resize(n+1); for(int i=0;i<=n;i++) f[i]=i;
    }
    int get(int x) {
        if(f[x]==x) return x;
        int fa=f[x];
        f[x]=get(fa);
        dis[x]+=dis[fa];
        return f[x];
    }
    bool merge(int x,int y,int val ){
        int fx=get(x), fy=get(y);
        if(fx==fy) return  false;
        dis[fx]=val+p[fy]-p[fx];
        f[fx]=fy;
        return true;
    }
};
void slove(){
    cin>>n;
    vector<array<int,3>>edge;
    vector<vector<pair<int,int>>>e(n+1);
    for(int i=1;i<=n;i++){
        for(int j=i+1;j<=n;j++) {
            cin>>A[i][j];
            edge.push_back({A[i][j],i,j } );
        }
    }
    sort(edge.begin(),edge.end(),[](const array<int,3>&a,const array<int,3>&b){
        return  a[0]<b[0];
    });
    DSU d;
    d.init(n+1);
    int edge_cnt=0;
    auto MST = [&]()->void{
        for(auto [w,u,v]:edge){
            if(d.get(u)==d.get(v) ) continue;
            d.merge(u,v,0);
            edge_cnt++;
            e[u].push_back({v,w});
            swap(u,v);
            e[u].push_back({v,w});
        }
    };
    MST();
    int root;
    vector< vector<int> >dis(n+1,vector<int>(n+1,inf));
    auto DFS = [&](auto&&self,int x,int fa,int cnt)->void{
        dis[root][x]=min(cnt,dis[root][x]);
        for(auto [u,w]:e[x]){
            if(u==fa) continue;
            self(self,u,x,cnt+w);
        }
    };
    if(edge_cnt!=n-1) {cout<<"No";return; }
    for(int i=1;i<=n;i++) root=i,DFS(DFS,i,-1,0);
    for(int i=1;i<=n;i++){
        for(int j=i+1;j<=n;j++){
            if(A[i][j]!=dis[i][j]) {cout<<"No";return ;}
        }
    }
    cout<<"Yes";return; 
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int t=1;
    //cin>>t;
    while(t--){
		slove();
    }
    return (0-0);
}
```
